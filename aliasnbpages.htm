AliasNbPages

AliasNbPages([string alias])

설명

총 페이지 갯수의 바꿈말을 정의합니다. 문서가 종료될 때 변환됩니다.

파라미터

alias
바꿈말. 기본값은 {nb}.

예제

class PDF extends FPDF
{
function Footer()
{
    //아래쪽에서 1.5 cm 위로 이동
    $this->SetY(-15);
    //글꼴을 Arial italic 8 로 선택
    $this->SetFont('Arial','I',8);
    //현재 페이지 번호와 총 페이지 갯수를 출력
    $this->Cell(0,10,'Page '.$this->PageNo().'/{nb}',0,0,'C');
}
}

$pdf=new PDF();
$pdf->AliasNbPages();
